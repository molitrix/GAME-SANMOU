        // --- 深色/浅色模式切换 ---
        const themeToggle = document.getElementById('theme-toggle');
        const themeIcon = document.getElementById('theme-icon');
        const htmlElement = document.documentElement;

        // 检查本地存储或系统偏好
        const prefersDark = window.matchMedia('(prefers-color-scheme: dark)');
        let currentTheme = localStorage.getItem('theme') ? localStorage.getItem('theme') : (prefersDark.matches ? 'dark' : 'light');

        // 应用初始主题
        function applyTheme(theme) {
            if (theme === 'dark') {
                htmlElement.classList.add('dark');
                themeIcon.classList.remove('fa-sun');
                themeIcon.classList.add('fa-moon');
            } else {
                htmlElement.classList.remove('dark');
                themeIcon.classList.remove('fa-moon');
                themeIcon.classList.add('fa-sun');
            }
            localStorage.setItem('theme', theme); // 保存用户选择
        }

        applyTheme(currentTheme);

        // 切换按钮事件
        themeToggle.addEventListener('click', () => {
            currentTheme = htmlElement.classList.contains('dark') ? 'light' : 'dark';
            applyTheme(currentTheme);
        });

        // 监听系统主题变化 (如果未手动设置过)
        prefersDark.addEventListener('change', (e) => {
            if (!localStorage.getItem('theme')) { // 仅当用户未手动设置时才跟随系统
                applyTheme(e.matches ? 'dark' : 'light');
            }
        });

        // --- 微信图标点击 ---

                    wechatPopup.classList.remove('show');
                }, 3000); // 3秒后自动隐藏
            }
        });

        // 点击弹出框以外区域隐藏
        document.addEventListener('click', (event) => {
            if (!wechatPopup.contains(event.target) && event.target !== wechatBtn && !wechatBtn.contains(event.target)) {

                    wechatPopup.classList.remove('show');
                    clearTimeout(popupTimeout); // 清除计时器
